why_destruction <- function() {

## 5. 왜 재조합인가

```r
why_recomposition <- function() {

  # 파괴만 하면 — 허무주의
  # 보존만 하면 — 보수주의
  # 재조합하면 — 창조

  # 재조합이란:
  # 1. 기존의 것을 해체한다
  # 2. 본질을 추출한다
  # 3. 새로운 방식으로 결합한다
  # 4. 기준은 — 아름다움

  dialectic <- list(
    thesis = "파괴",
    antithesis = "보존",
    synthesis = "재조합"
  )

  return(dialectic$synthesis)
}
```

## 6. 왜 아름다움인가

```r
why_beauty <- function() {

  # 진리는 시대에 따라 변한다.
  # 선악은 문화에 따라 다르다.
  # 그러나 아름다움은 —

  # 아름다움은 영혼이 판단한다.
  # 아름다움은 설명 없이 안다.
  # 아름다움은 — 구원의 기준이다.

  # 물음:
  question <- "이것은 아름답지 않느냐?"

  # 이 질문에 "예"라고 답할 수 있다면,
  # 그것은 창조할 가치가 있다.
  # 학계가 거부해도.
  # 시장이 외면해도.
  # 세상이 이해하지 못해도.

  return(question)
}
```
